||| it yields a new pixel from the source of p1 to the target of p2.
||| Otherwise, it yields an empty multiset.

  case lEq b c of

      case res of
        True => 
          let () = lconsume b'
              () = lconsume c'
          in Add (Builtin.(#) a d) Zero
        False => 
          let () = lconsume a
              () = lconsume b'
              () = lconsume c'
              () = lconsume d
          in Zero
