  compare (Add x xs) Zero = let () = consume x; () = consume xs in GT
  compare (Add x xs) (Add y ys) =
    let (x1, x2) = copy x
        (y1, y2) = copy y 
    in case compare x1 y1 of
         EQ  => let () = consume x2; () = consume y2 in compare xs ys
         res => let () = consume x2; () = consume y2;
